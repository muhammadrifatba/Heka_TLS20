#include <stdio.h>
#include <stdlib.h>
struct vol {
    int p;
    int l;
    int t;
};

int main(){
    struct vol *ptr;
    int i, n, total;
    printf ("Jumlah kotak yang dihitung : ");
    scanf ("%d", &n);
    printf ("\n");
    if (n<=100 && n>=1){

    ptr = (struct vol *)malloc(n * sizeof(struct vol));
    for (i = 0; i < n; ++i) {
        printf ("Kotak ke-%d\n", i+1);
        printf ("Masukan panjang, lebar, dan tinggi kotak (berurutan) : \n");
        printf ("(Pisahkan dengan spasi)\n");
        scanf ("%d %d %d", &ptr->p, &ptr->l, &ptr->t);
        if (ptr->p <= 100 && ptr->p >= 1){
            if (ptr->l <= 100 && ptr->l >= 1){
                if (ptr->t <= 100 && ptr->t >= 1){
                    if (ptr->t < 41){
                        total=ptr->p*ptr->l*ptr->t;
                        printf ("Maka volume kotak %d : \n", i+1);
                        printf ("%d x %d x %d = %d \n\n", ptr->p, ptr->l, ptr->t, total);
                    }
                    else
                        printf ("Tinggi %d melewati batas!! \n\n", ptr->t);
                }
                else if (ptr->t < 1){
                    printf ("Setidaknya tinggi bernilai satu\n\n");
                }
                else {
                    printf ("Nilai tinggi terlalu besar\n\n");
                }
                        
            }
            else if (ptr->l <1){
                printf ("Setidaknya lebar bernilai satu\n\n");
            }
            else {
                printf ("Nilai lebar terlalu besar\n\n");
            }
        }
        else if (ptr->p < 1){
            printf ("Setidaknya panjang bernilai satu\n\n");
        }
        else {
            printf ("Nilai panjang terlalu besar\n\n");
        }
    }
    } 
    if (n<1) {
        printf ("Setidaknya nilai yang dihitung satu\n\n");
    }
    if (n>100) {
        printf ("Nilai terlalu besar\n\n");
    }
    return 0;
}
